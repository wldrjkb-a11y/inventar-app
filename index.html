<script type="module">


/* 🔥 FIREBASE IMPORTS */

import { initializeApp } from "https://www.gstatic.com/firebasejs/10.7.1/firebase-app.js";

import {

  getAuth,

  createUserWithEmailAndPassword,

  signInWithEmailAndPassword,

  signOut,

  onAuthStateChanged

} from "https://www.gstatic.com/firebasejs/10.7.1/firebase-auth.js";


import {

  getFirestore,

  collection,

  addDoc,

  getDocs,

  query,

  where

} from "https://www.gstatic.com/firebasejs/10.7.1/firebase-firestore.js";


/* 🔥 FIREBASE CONFIG */

const firebaseConfig = {

  apiKey: "DEIN_API_KEY",

  authDomain: "inventar-appjw.firebaseapp.com",

  projectId: "inventar-appjw",

  storageBucket: "inventar-appjw.appspot.com",

  messagingSenderId: "156223083174",

  appId: "1:156223083174:web:599dbed39d7d2a69bc9a67"

};


const app = initializeApp(firebaseConfig);

const auth = getAuth(app);

const db = getFirestore(app);


let items = [];


/* 🔐 USERNAME → EMAIL */

function toEmail(u){

  return u.trim().toLowerCase() + "@app.local";

}


/* ---------------- AUTH ---------------- */


window.register = async function(){

  let u = username.value;

  let p = password.value;

  await createUserWithEmailAndPassword(auth, toEmail(u), p);

};


window.login = async function(){

  let u = username.value;

  let p = password.value;

  await signInWithEmailAndPassword(auth, toEmail(u), p);

};


window.logout = async function(){

  await signOut(auth);

};


/* ---------------- SAVE ITEM ---------------- */


window.saveItem = async function(){


  const user = auth.currentUser;

  if(!user) return;


  await addDoc(collection(db,"items"), {

    uid: user.uid,

    name: name.value,

    cat: cat.value,

    stock: parseInt(stock.value) || 0,

    min: parseInt(min.value) || 0

  });


  name.value = "";

  stock.value = "";

  min.value = "";


  load();

};


/* ---------------- LOAD ---------------- */


async function load(){


  const user = auth.currentUser;

  if(!user) return;


  const q = query(

    collection(db,"items"),

    where("uid","==",user.uid)

  );


  const snap = await getDocs(q);


  items = snap.docs.map(d => ({

    id: d.id,

    ...d.data()

  }));


  render();

}


/* ---------------- RENDER ---------------- */


function render(){


  list.innerHTML = "";


  items.sort((a,b)=>a.name.localeCompare(b.name));


  items.forEach(i => {


    list.innerHTML += `

      <div class="item">

        <b>${i.name}</b><br>

        Kategorie: ${i.cat}<br>

        Bestand: ${i.stock} | Min: ${i.min}

      </div>

    `;


  });


}


/* ---------------- AUTH STATE ---------------- */


onAuthStateChanged(auth, user => {


  if(user){

    loginBox.style.display = "none";

    app.style.display = "block";

    load();

  } else {

    loginBox.style.display = "block";

    app.style.display = "none";

  }


});
